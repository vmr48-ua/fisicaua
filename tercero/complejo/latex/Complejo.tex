\chapter{El cuerpo de los números complejos}
  \section{Definiciones básicas}
    \definicion{Número complejo}{
      Un \textbf{número complejo} $z$ es un par ordenado de números reales $a,b$ escrito como
      $\boxed{z=\left(a,b\right)}$ en coordenadas cartesianas. Existe una notación equivalente,
      la forma binómica: $\boxed{z=a+ib}$ siendo $i=\left(0,1\right)$. \\

      El conjunto de los número complejos se denota por: $C:=\left\{(a,b):a,b\in\bbR\right\}$
    }
    \comentario{
      Siempre que $a=0$ sea un número imaginario puro, y $b=0$ sea un número real.
    }
    \definicion{Conjugado}{
      Llamamos conjugado de un número complejo al número denotado $\boxed{\ovl{z}=a-ib}$, siendo
      $z=a+ib$. Geométricamente, podemos decir que el eje real actúa de 'espejo' del número en el plano.
    }
    \comentario{
      Llamamos $\bbC$ al cuerpo de los numeros complejos. $\bbC$ es un cuerpo conmutativo, pero no totalmente ordenado. En cambio, cualquier ecuación algebraica
      tiene solución en los complejos. De todas formas, el teorema fundamental del álgebra nos asegura que tendrá n soluciones en los complejos
    }
    \comentario{
      Cuando los coeficientes de una ecuación algebraica son reales, las soluciones complejas vienen por pares.
    }
    \teorema{Operaciones elementales}{
      \renewcommand{\arraystretch}{1.6}
      \begin{tabular}{lll}
        \textbf{SUMA}& $\left(a+bi\right)+\left(c+di\right)=\left(a+c\right)+\left(b+d\right)i$\\
        \textbf{RESTA}& $\left(a+bi\right)-\left(c+di\right)=\left(a-c\right)+\left(b-d\right)i$\\
        \textbf{PRODUCTO}& $\left(a+bi\right)\cdot\left(c+di\right)=\left(ac-bd\right)+\left(ad+bc\right)i$ &(teniendo en cuenta que $i^2=-1$)\\
        \rule{0pt}{2.3em}\textbf{DIVISIÓN}& $\cfrac{a+bi}{c+di}\ =\ \cfrac{a+bi}{c+di}\cdot\cfrac{c-di}{c-di}\ =\ \cfrac{ac+bd}{c^2+d^2}+\left(\cfrac{bc-ad}{c^2+d^2}\right)i$ &(multiplicando por el conjugado)\\
      \end{tabular}
    }

      El producto de un número complejo por su elemento inverso es la unidad.
    }

    \definicion{Función armónica conjugada}{
      Sea $u\colon\mcD\subset\bbR^2\rightarrow\bbR$ una función armónica en un abierto
      de $\mcD\subset\bbR^2$ diremos que $v\colon\mcD\subset\bbR^2\rightarrow\bbR$ es
      una \textbf{función armónica conjugada} de $u$ en $\mcD$ si $v$ es armónica en
      $\mcD$ y satisfacen las condiciones de \textit{Cauchy-Riemann}, (o equivalentemente
      la función $f(x+iy)=u(x,y)+iv(x,y)$ es holomorfa en $\left\{x+iy\in\bbC\colon (x,y)
      \in\mcD\right\}$)
    }
    \comentario{
      Una función armónica es aquella que satisface la ecuación de Laplace.
    }
    \teorema{}{
      Sea $u(x,y)\colon\mcD\rightarrow\bbR$ es una función armónica de $\mcD$ y
      consideramos $v$ una región rectangular contenida en $\mcD$. Entonces existe una
      conjugada armónica de $u(x,y)$ en $v$.
    }

    \subsection{Función potencia}
      \definicion{Potencia de exponente arbitrario}{
        Sea $z\in\bbC \setminus {0}$ y $\alpha\in\bbC$, tomamos por definición $z^\alpha$, llamada
        potencia de exponente arbitrario como el conjunto de todos los valores dados por:
        \begin{equation}
          z^\alpha = e^{\alpha\log z}
        \end{equation}
        Dohde $\log z$ representa el conjunto de todos los logaritmos de $z$.
      }
      \definicion{Función exponencial general}{
        Sea $a\in\bbC\setminus{0}$, tomaremos por definición $a^z$, llamada función 
        exponencial general como el conjunto de todos los valores dados por:
        \begin{equation}
          a^z = \exp\left(z\log a\right)
        \end{equation}
        Donde $\log a$ es el conjunto de todos los logaritmos de $a$.
      }
      \ejemplo{}{
        \begin{itemize}
          \item $(-2)^{\frac12}=e^{\frac12 \log(-2)}=e^{\frac12\left(\ln 2+i(\pi+2\pi k)\right)}$,
                con $k\in\bbZ$ (tomando la primera definición, $z=-2$ y $\alpha=\frac12$)
          \item $2^i=e^{i\log(2)}=e^{i(\ln 2+2\pi ki)}=e^{-2\pi k}e^{i\ln 2}=
                e^{-2\pi k\left(\cos(\ln 2)+i\sin(\ln 2)\right)}$ con $k\in\bbZ$
          \item $(-1)^{\frac{1}{\pi}}= e^{\frac{1}{\pi}\log(-1)}= e^{\frac{1}{\pi}(\pi+2\pi k)i}
                =e^i\cdot e^{2ki}=e^{(2k+1)i}$, con $k\in\bbZ$
        \end{itemize}
      }

  \section{Preliminares topológicos}
    \definicion{Entorno perforado}{
      Llamamos \textbf{entorno perforado} de un punto $z_0\in\bbC$ a un abierto de la forma
      $\left\{z\in\bbC\colon 0<\abs{z-z_0}<\epsilon\right\}$, con $\epsilon >0$
    }
    \definicion{Tipos de conjuntos}{
      \begin{itemize}
        \item Diremos que dos conjuntos $A$ y $B$ de $\bbC$ están \textbf{espaciados} 
              si $\ovl{A}\cap B=A\cap\ovl{B}=\varnothing$\\ Siendo $\ovl{A}$ la clausura,
              la adherencia de $A$.
        \item Diremos que un conjunto del plano es \textbf{conexo} si no puede ser
              escrito como unión de dos subconjuntos no vaciós y separados.
      
        \item Diremos que un conjunto $P\in\bbC$ es \textbf{poligonalmente conexo} si cada
              par de puntos de $P $ pueden ser unidos mediante una poligonal contenida en $P$.
              (una ponigonal es una unión finita de segmentos).
        \item Un conjunto $E\in\bbC$ es \textbf{estrellado} si existe un punto $a\in E$
              tal que $\left[a,z\right]\subset E \hspace{0.5cm}\forall z\in E$
        \item Un conjunto $C\in\bbC$ es \textbf{convexo} si $\left[z,w\right]\subset C 
              \hspace{0.5cm}\forall z,w\in C$ (cualquier segmento formado por puntos
              del conjunto está dentro del conjunto).

              \textit{Nota: Sea $U\in\bbC$ un conjunto abierto, entonces $U$ es conexo si y sólo
              si es poligonalmente conexo.}
        \item Llamamos \textbf{simplemente conexo} al conjunto $S\subset \bbC$ del cual cada 
              curva cerrada simple (sin autointersecciones) en $S$ puede contraerse dentro
              del conjunto hasta ser un punto (no tiene agujeros).
      \end{itemize}
    }
  \section{Integración sobre caminos}
    \definicion{Curva}{
      Llamamos \textbf{curva} a una aplicación continua $\gamma\colon\left[a,b\right]
      \rightarrow\bbC$ con $a<b$, tal que a un número real $t\in\left[a,b\right]$ le
      corresponde un número complejo $\gamma(t)=x(t)+iy(t)$ donde $x(t)$ e $y(t)$ 
      son funciones reales y continuas.
      \begin{itemize}
        \item La \textbf{traza o trayectoria} de la curva $\gamma(\left[a,b\right])=
              \left\{\gamma(t)\colon a\leq t\leq b\right\}$ será representado por 
              $\gamma^*$
        \item Diremos que la curva es \textbf{cerrada} cuando $\gamma(a)=\gamma(b)$
      \end{itemize}
    }
    \definicion{Camino}{
      Una curva $\gamma\colon\left[a,b\right]\rightarrow\bbC$ es \textbf{diferenciable}
      cuando $\gamma$ es derivable en todo punto de $\left[a,b\right]$. 
      Una curva $\gamma\colon\left[a,b\right]\rightarrow\bbC$ es \textbf{suave} si es
      diferiencable (o de clase $\mcC^1(\left[a,b\right])$), si $\gamma$ es
      derivable en $\left[a,b\right]$ y su derivada es continua.
      Llamamos \textbf{camino} a una curva suave a trozos (diferenciable con continuidad
      a trozos).
    }
    \comentario{
      Los casos de curvas rectificables (aquellas curvas parametrizadas por $\gamma(t)
      =x(t)+iy(t)$), con $t\in[a,b]$,  para las que existe:
      \[\sup\left\{\sum_{j=1}^{n}\sqrt{(x(t_j)-x(t_{j-1}))^2+(y(t_j)-y(t_{j-1}))^2}\right\}\]
      Con $P$ en el conjunto de posibles particiones de $\left[a,b\right]$. En estos
      casos, la longitud de la curva se calcula como: \[L(\gamma)=\displaystyle\int_a^b
      \abs{\gamma^\prime(t)}\ dt=\int_a^b\sqrt{(x^\prime(t)^2+y^\prime(t)^2)}\ dt\]
    }
    \definicion{Integral compleja}{
      Sea $\gamma\colon\left[a,b\right]\rightarrow\bbC$ un camino y $f$ una función 
      continua en $\gamma^*\in\bbC$, definimos la \textbf{integral conmpleja} de $f$
      a lo largo $\gamma$ por:
      \[\int_{\gamma}f(z)\ dz=\int_a^bf\left(\gamma(t)\right)\cdot\gamma^\prime(t)\ dt\]
      \textit{Nota: Cuando no se diga nada, se supondra que el sentido de recorrido 
      sobre un camino cerrado será el antihorario.}
    }
    \ejemplo{$\int_\gamma \ovl{z}\ dz$, donde $\gamma^*$ es el segmento $[1+i,2+4i]$}{
      $\gamma(t)=(1-t)(1+i)+t\cdot(2+4i) \Longrightarrow \gamma^\prime(t)=-(1+i)+2+4i=
      1+3i\hspace{0.3cm}$ con $t\in[0,1]$.\\

      Si separamos en parte real y imaginaria nos queda:\\
      $\gamma(t)=1+t+i(1+3t)\Longrightarrow\int_\gamma\ovl{z}\ dz=\int_0^1(1+t
      -i(1+3t))(1+3i)\ dt=\int_0^1(1+t-i(1+3t)+3i+3it+3+9t)\ dt=\int_0^1 (4+10t+2i)\ dt
      =\left[(4+2i)t+5t^2\right]^1_0=4+2i+5=9+2i$
    }

    une $1+i$ con $2+4i$}{

    \teorema{Propiedades básicas de integrabilidad}{
      Sea $\gamma\colon[a,b]\rightarrow\bbC$ un camino, y consideremos dos funciones
      continuas $f$ y $g$ continuas en $\gamma^*$, entonces se cumplen las siguientes
      propiedades:
      \begin{enumerate}
        \item $\int_\gamma \left(f(z)+g(z)\right)\ dz=\int_\gamma f(z)\ dz+\int_\gamma g(z)\ dz$
        \item $\int_\gamma cf(z)\ dz=c\int_\gamma f(z)\ dz\hspace{0.3cm}\forall c\in\bbC$ 
        \item $\int_{-\gamma} f(z)\ dz= -\int_\gamma f(z)\ dz$, donde $-\gamma$ es el camino
              opuesto a $\gamma$: $(-\gamma)(t)=\gamma(b+a-t)$ con $t\in[a,b]$
        \item Si $\beta$ es otro camino tal que $\gamma+\beta$ está definido (el punto inicial
              de $\beta$ coincide con el punto final de $\gamma$) y la función $f$ es también 
              continua en $\beta^*$ y $(\gamma+\beta)^*$, entonces:\\
              $\int_{\beta+\gamma}f(z)\ dz=\int_\gamma f(z)\ dz+\int_\beta f(z)\ dz$
        \item La integral sobre un camino es invariante bajo una parametrización.
      \end{enumerate}
    }

    \teorema{Teoremas de Cauchy para determinados tipos de regiones}{
      \subsubsection*{Teorema de Cauchy para triángulos}
        Sea $f\colon U\longrightarrow\bbC$ una función holomorfa en un conjunto abierto $U\in\bbC$,
        tenemos que para cualquier triángulo $\Delta (\Delta=\text{int}(\Delta))\cup F_r(\Delta)$ 
        contenido en $U$:
        \begin{equation}
          \int_\gamma f(z)\ dz=0\hspace{1cm}\text{donde }\gamma=F_r(\Delta)
        \end{equation}\\

      \subsubsection*{Teorema de Cauchy para regiones estrelladas}
        Consideremos $f\colon U\longrightarrow \bbC$ una función holomorfa en un conjunto
        estrellado y abierto de $U\subset\bbC$, entonces:
        \begin{equation}
          \int_\gamma f(z)\ dz=0
        \end{equation}
        para cualquier camino cerrado $\gamma$ contenido en $U$.\vspace{0.3cm}\\

      \subsubsection*{Teorema local de Cauchy}
        Sea $f\colon\mcD(z_0,r)\longrightarrow\bbC$ una función holomorfa en un disco 
        abierto $\mcD(z_0,r)$ para algún $z_0\in\bbC$ y $r>0$, entonces:
        \begin{equation}
          \int_\gamma f(z)\ dz=0
        \end{equation}
        para todo camino $\gamma\text{cerrado en }\mcD(z_0,r)$

        \textit{\textbf{Nota:} No siempre la integral sobre un camino cerrado es cero,
        por ejemplo: $\int_{(0,1)}\frac{1}{z}\ dz=\int_0^{2\pi}\frac{1}{e^{it}}ie^{it}\ dt=2\pi i\neq 0$ 
        ya que el disco no es estrellado porque tiene un agujero en el origen (no es 
        simplemente conexo).}\vspace{0.3cm}\\

      \subsubsection*{Teorema de Cauchy extendido para triánglos}
        Sea $f\colon U\longrightarrow\bbC$ una función contitua en un conjunto abierto $U\in\bbC$,
        y holomorfa en $U\setminus\left\{z_0\right\}$ para algún $z_0\in U$.
        Para cualquier triángulo $\Delta (\Delta=\text{int}(\Delta))\cup F_r(\Delta)$ 
        contenido en $U$:
        \begin{equation}
          \int_\gamma f(z)\ dz=0\hspace{1cm}\text{donde }\gamma=F_r(\Delta)
        \end{equation}\\

      \subsubsection*{Teorema de Cauchy para conjuntos convexos}
        Sea $U$ un conjunto convexo y $f\colon U\longrightarrow\bbC$ una función continua
        en tono $U$ y holomorfa en $U\setminus\left\{z_0\right\}$ para algún $z_0\in U$, 
        entonces:
        \[\int_\gamma f(z)\ dz=0\]
        para cualquier $\gamma$ camino cerrado en $U$.
        \textit{\textbf{Nota:}Puede ser holomorfa también en $z_0$ de forma más general.}
    }
    \corolario{}{
      Sea $U$ un conjunto convexo y $f\colon U\longrightarrow\bbC$ una función holomorfa
      en $U$ entonces:\\

      $F(z)=\int_{\gamma_z} f(s)\ ds$ con $z\in U$ es holomorfa en $U$ y $F^\prime(z)=f(z)$ 
      $\forall z\in U$\\
      donde $\gamma_z$ es un camino en $U$ que une $z_0$ con $z$, siendo
      $z_0$ un punto prefijado de $U$.
    }

    \definicion{Índice de un camino cerrado}{
      Sea $\gamma$ un camino cerrado y $z_0\in\gamma^*$, se define el \textit{índice de }$z_0$ 
      respecto del camino $\gamma$ y se denota por $n(\gamma,z_0)$:
      \[n(\gamma, z_0)=\dfrac{1}{2\pi i}\int_{\gamma}\dfrac{dz}{z-z_0}\]
      Geométricamente, $n(\gamma,z_0)$ representa el número de vueltas completas en sentido 
      antihorario que da un camino cerrado $\gamma$ alrededor de $z_0$.

      \[\implies n(\gamma_1,z_0)=\dfrac{1}{2\pi i}\displaystyle\int_{\gamma_1}\dfrac{dz}{z-z_0}=

      \begin{enumerate}
        \item El índice $n(\gamma, z_0)$ es un número entero.

        \item $n(-\gamma,z_0)=-n(\gamma,z_0)$

        \item $n(\gamma,z)=0$ $\forall z$ en la componente conexa no acotada de $\bbC\setminus\gamma^*$
      \end{enumerate}

      valor del parámetro $r>0$, se deduce que $\abs{f^\prime(z)}\leq\nicefrac{M}{r}$, $z\in\bbC$
      Dado que $M$ no depende de $r$, se deduce que $f^\prime(z)=0$ $\forall z\in\bbC\implies
      f(z)=cte$
    }
    \teorema{Fundamental del Álgebra}{
      Todo polinomio $p(z)$ no constante tiene al menos una raíz.\\
      Donde $p(z)=a_n z^n+a_{n-1}z^{n-1}+\dots+a_1 z+a_0$

    \comentario{
      \textbf{Prueba:}
      Por Reducción al Absurdo, supongamos que $p(z)$ no tiene raíces.\\
      Entonces $f(z)=\dfrac{1}{p(z)}$ es una función entera (holomorfa en $\bbC$).

      $\Longrightarrow \exists M>0\colon\abs{p(z)}>1\forall z\colon\abs{z}>M\Longrightarrow
      \abs{f(z)}=\dfrac{1}{\abs{p(z)}}<1\forall z\colon\abs{z}>M\Longrightarrow 
      \text{ f acotada en }\bbC$\\
      Y por el teorema de Liouville, $f\equiv$ cte
    }

    \teorema{Lema de Schwarz}{
      Sea $f\colon D(0,1)\rightarrow \bbC$ una función holomorfa en $D(0,1)$. 
      Supongamos que $f(0)=0$ y $\abs{f(z)}\leq 1$ $\forall z\in D(0,1)$, entonces
      $\abs{f(z)}\leq\abs{z}$ y $\abs{f^\prime(0)}\leq 1$.\\

      Además, si existe $z_0\in D(0,1)\setminus \left\{0\right\}$ tal que
      $\abs{f(z_0)}=\abs{z_0}$ o se cumple que $\abs{f^\prime(0)}=1$, entonces
      $f(z)=az$ $\forall z\in D(0,1)$ para algún $a\in\bbC$ tal que $\abs{a}=1$
    }

    }

      \[\text{Re }Q_z(t)=P_r(\theta-t)=\dfrac{R^2-r^2}{\abs{Re^{it}-z}^2}\]

    \definicion{Fórmula Integral de Poisson}{
      Sea $f(z)$ analítica en $D(z_0,R)$ y continua en $\ovl{D}(z_0,R)$, para algún
      $z_0\in\bbC$ y $R>0$. Para cada $z=z_0+re^{i\theta}\in D(z_0,R), 0\leq r< R$, 
      $0\leq\theta<2\pi$ se tiene que
      \[f(z)=\dfrac{1}{2\pi}\int_{0}^{2\pi}P_r(\theta-t)f(z_0+Re^{it})\ dt\]

\chapter{Series de Potencias}
  \noindent Si $f$ es una función holomorfa, podemos expresar (almenos localmente) como
  un polinomio de grado infinito. 
  \[f(z)=\sum_{n=0}^{\inf}a_n(z-z_0)^n\]
  \section{Notación}
    \noindent La serie $\displaystyle \sum_n z_n$ es la sucesión
    %$\left\{\displaystyle\sum_{k=1}^n z_k\right\}=\left\{z_1,z_1+z_2,\dots,z_1+\dots+\z_n,\dots\right\}$
    %En el caso en que la serie sea convergente, escribimos $\lim_{n\rightarrow\inf}\sum_{n=1}z_n=\displaystyle\sum_{n=1}^\inf z_n$
    \definicion{Convergencia de Series}{
      \begin{itemize}
        \item Una sucesión $\left\{z_n\right\}\in\bbC$ converge a $z\in\bbC$ si 
              $\forall\varepsilon>0$ $\exists N\in\bbN\colon \in\bbN\Rightarrow\abs{z_n-z}<\varepsilon$

        \item Una serie $\sum_{n=1}z_n$ converge a $z\in\bbC$ si $\forall\varepsilon>0$ 
              $\exists N\in\bbN\Rightarrow\abs{z_1+\dots+z_n-z}<\varepsilon$\\

              Como $\abs{z_n-z}\in\bbR$, es equivalente la covergencia de sucesiones en números
              reales que en números complejos.

        \item Una serie $\sum_{n=1}z_n$ converge absolutamente si $\sum_{n=1}^\infty\abs{z_n}<\infty$
      \end{itemize}
    }
    \comentario{
      Si $z_n=x_n+iy_n$, entonces $\displaystyle\sum_{n=1}z_n$ converge $\Longleftrightarrow\displaystyle\sum_{n=1}x_n$ y $\displaystyle\sum_{n=1}y_n$
      converge y en ese caso, $\displaystyle\sum_{n=1}^\infty z_n=\displaystyle\sum_{n=1}^\infty x_n+i\displaystyle\sum_{n=1}^\infty y_n$\\

      Convergencia absoluta $\Longrightarrow$ Convergencia.
    }
    \comentario{
      \subsubsection*{Recordatorio: Seriees de números positivos}
      Sea $\displaystyle \sum_{n=1}\rho_n$, $\rho_n\geq 0$. Sean
      $A=\text{limsup}\left(\rho_n\right)^{\nicefrac{1}{n}}$ y 
      $B=\text{limsup}\left(\dfrac{\rho_{n+1}}{\rho_n}\right)^{\nicefrac{1}{n}}$ Entonces:
      \begin{itemize}
        \item Si $A<1$ ó $B<1$, $\displaystyle\sum_{n=1}^\infty\rho_n<\infty$
        \item Si $A>1$ ó $B>1$, $\displaystyle\sum_{n=1}^\infty\rho_n=\infty$
      \end{itemize}
      $\text{limsup}(x_n)=\lim\text{sup}\left\{x_k\colon k\geq n\right\}$ Si $\exists\lim x_n$, 
      $\text{lumsup} x_n=\lim x_n$\hspace{0.5cm}$\text{limsup}(x_n y_n)=\lim x_n \text{limsup} y_n $
      si $\exists\lim x_n$
    }
    \ejemplo{$\displaystyle\sum_{n=1}n(1+i)^n(2i)^{-n}$}{
      $a_n=n(1+i)^n(2i)^{-n}\Longrightarrow \abs{a_n}=n\abs{1+i}^n\abs{2i}^{-n}
      =\dfrac{n(\sqrt{2})^n}{2^n}$\\
      $\sqrt[n]{\abs{a_n}}\sqrt[n]{n}\dfrac{\sqrt{2}}{2}\longrightarrow
      \dfrac{\sqrt{2}}{2}\Longrightarrow \text{limsup}\sqrt[n]{\abs{a_n}}=\dfrac{\sqrt{2}}{2}<1\Longrightarrow$
      La serie converge absolutamente.\\

      En general, si la sucesión es convergente, podemos pasar una sucesión de 
      números naturales a reales para aplicar los criterios de los reales.
    }
    \ejemplo{$\displaystyle\sum_{n=1}n^2 2^n(3+i)^{-n}$}{
      $a_n=n^2 2^n(3+i)^{-n}\Longrightarrow \abs{a_n}=\dfrac{n^2\cdot 2^n}{(\sqrt{10})^n}$\\
      $\dfrac{a_{n+1}}{\abs{a_n}}=\dfrac{(n+1)^2\cdot 2^{n+1}}{(\sqrt{10})^{n+1}}\cdot
      \dfrac{(\sqrt{10})^n}{n^2\cdot 2^n}=\left(\dfrac{n+1}{n}\right)^2\dfrac{2}{\sqrt{10}}
      \longrightarrow \dfrac{2}{\sqrt{10}}<1$
    }
    \ejemplo{Serie geométrica: $\displaystyle\sum_{n\geq 0}z^n$}{
      $a_n=z^n\Longrightarrow\abs{a_n}=\abs{z}^n$\hspace{1cm}$\dfrac{z^{n+1}-1}{z-1}=1+z+\dots+z_n$\\
      Si $z=1$, la serie no converge, si $z\neq1$:
      $\displaystyle\sum_{n\geq 0}z^n=\left\{1+\dots+z_n\right\}_{n\in\bbN U \left\{0\right\}}=
      \left\{\dfrac{z^{n+1}-}{z-1}\right\}$\\
      \begin{itemize}
        \item Si $\abs{z}<1, z^{n+1}\longrightarrow 0\Longrightarrow\displaystyle\sum_{n=0}^\infty z^n=\dfrac{1}{1-z}$
        \item Si $\abs{z}\geq 1, z^{n+1}$ no converge $\Longrightarrow$ La serie no converge.
      \end{itemize}
    }
    \definicion{Series de potencias}{
      Sean $z_0\in\bbC, \left\{a_n\right\}\in\bbC$. La serie de potencias de coeficientes
      $\left\{a_n\right\}$ y otro $z_0$ es la serie de funciones $\displaystyle\sum_{n\geq 0}a_n
      \left(z-z_0\right)^n$\\

      Decimos que la serie converge (puntualmente) en $S\subseteq\bbC$ si:
      \begin{equation}
        \sum_{n\geq 1}a_n(w_0-z_0)^n \text{ converge }\forall w_0\in S
      \end{equation}

      Decimos que la serie converge uniformemente a $S\subseteq\bbC$ si:
      \begin{equation}
        \forall z>0 \exists N\in\bbN\colon n\geq N\Longrightarrow
        \abs{\sum_{k=1}^n a_k(w_0-z_0)^k-\sum_{k=1}^\infty a_k(w_0-z_0)^k}
        <\varepsilon\hspace{0.4cm}\forall w_0\in S
      \end{equation}
    }
    \definicion{Teorema de Abel}{
      Sea $\displaystyle\sum_{n\geq 0}a_n(z-z_0)^n$ y sea $z_1\in\bbC$ tal que
      la serie converge puntualmente en $z_1$. Si $r=\abs{z_0-z_1}$, entonces:
      \begin{itemize}
        \item La serie converge \textbf{absolutamente} en cada punto de $D(z_0,r)$
        \item La serie converge \textbf{uniformemente} en cada compacto de $D(z_0,r)$
      \end{itemize}
    }
    \definicion{Radio de convergencia de una serie de potencias}{
      Sea $\displaystyle\sum_{n\geq 0}a_n(z-z_0)^n$ una serie de potencias, definimos el radio
      de convergencia de la serie como:

      \[r=\left(\text{limsup}(\abs{a_n}^{\nicefrac{1}{n}})\right)^{-1}\]

      Estudiando $(+\infty)^{-1}=0,\hspace{0.5cm}0^{-1}=+\infty$
    }
    \teorema{Teorema de convergencia de las series de potencias}{
      Sea $\displaystyle\sum_{n\geq 0}a_n(z-z_0)^n$ y sea $r=\left(\text{limsup}\abs{a_n}^{\nicefrac{1}{n}}\right)^{-1}$, entonces:
      \begin{itemize}
        \item La serie converge absolutamente en $D(z_0,r)$
        \item Converge uniformemente en cada compacto de $D(z_0,r)$, y
        \item No converge en ningún punto $z\in\bbC\setminus\ovl{D}(z_0,r)$
      \end{itemize}
    }

\chapter{Teoría de Residuos}
  \section{Introducción}
    \noindent Objetivo: 
      \[\int_\gamma f(z)\ dz \text{ con }\gamma\text{ camino cerrado}\]
      \[\int_\gamma \dfrac{f(w_0)}{(z-w)^n}\ dw=2\pi i\dfrac{f^{(n)}(z)}{(n+1)!}\]
    Si $f(z)=\displaystyle \sum_{n=\infty}^{+\infty}a_n(z-z_0)^n\ $ $\ \forall z\in A(z_0,0,R)
    \Longleftrightarrow a_n=\dfrac{1}{2\pi i}\displaystyle\int_{C(w_0,r)}\dfrac{f(w)}{(w-z_0)^{n+1}}\ dz$
    $\Longrightarrow a_{-1}=\dfrac{1}{2\pi i}\displaystyle\int_{C(w_0,r)}f(w_0)\ dw$

    \ejemplo{$\displaystyle\int_{C(1,1)}\left(\dfrac{1}{(z-1)^2}+\dfrac{3}{z-1}\right)\ dz$}{
      $\displaystyle\int_{C(1,1)}\left(\dfrac{1}{(z-1)^2}+\dfrac{3}{z-1}\right)\ dz = a_{-1}\cdot 2\pi i=6\pi i$
    }
    \ejemplo{$\displaystyle\int_{C(w,r)}\dfrac{f(w)}{(w-z)}\ dw$}{
      $=2\pi i a_{-2}$ (de f)\\
      $=\displaystyle\int_{C(z,r)}g(w)\ dw=2\pi i b_{-1}$ (de g)
    }
    \definicion{Residuo}{
      Sea $z_0\in\bbC$ una singularidad aislada (para poder tener desarrollo de Laurent) de $f$,
      y sea $\displaystyle\sum_{n=-\infty}^{+\infty}a_n(z-z_0)^n$ la serie de Laurent de $f$ centrada
      en $z_0$, entonces el residuo de $f$ en $z_0$ es:

      \begin{equation}
        \text{Res}(f,z_0)=a_{-1}
      \end{equation}
    }

    \noindent Lo interesante es que para calcular integrales, sólo necesitamos el residuo, no nos hace falta toda la serie de Laurent.
    
    \comentario{
      Si $c$ es un camino cerrado en el anillo de convergencia de la serie de Laurent
      y es homóloga a $C(z_0,r)$, entonces:

      \begin{equation}
        \int_{\gamma}f(w)\ dw=2\pi i \text{Res}(f,z_0)
      \end{equation}
    }
    \teorema{Cálculo de residuos}{
      Clasificamos la singularidad de $f$ en $z_0$:
      \begin{itemize}
        \item Si no hay singularidad o la singularidad es evitable, la serie de Laurent 
              coincide con la serie de Taylor $\sum_{n=-\infty}^{+\infty}a_n(z-z_0)^n=\sum_{n=0}^{+\infty}a_n(z-z0)^n
              \Longrightarrow a_{-1}=0\Longrightarrow \text{Res}(f,z_0)=0$
        \item Si $z_0$ es un polo de oden $m$, $\sum_{n=-\infty}^{+\infty}a_n(z-z_0)^n=\sum_{n=-m}^{+\infty}a_n(z-z_0)^n$
              si $g(z)=(z-z_0)^n f(z)$, entonces $g(z)=\sum_{n=0}^{\infty}a_{n-m}(z-z_0)^n$ es un desarrollo de Taylor.\\
              $\Longrightarrow a_{-1}=\dfrac{g^{m-1}(z_0)}{(m-1)!}=\dfrac{\lim_{z\rightarrow z_0}g^{(m-1)}(z)}{(m-1)!}$
        \item Si $z_0$ es una singularidad esencial, calcelamos el desarrollo de Laurent.
      \end{itemize}
    }
    \ejemplo{Calcula el residuo en 1 y en 0 de esta función $f(z)=\dfrac{e^z}{z(z-9)^2}$}{
      $\text{Res}(f,z)=0\ \forall z\in\bbC\setminus\left\{0,9\right\}$\\

      $z=0$ es un polo de orden 1 $\Longrightarrow \lim_{z\rightarrow 0} (z-0)f(z)=
      \lim_{z\rightarrow 0}\dfrac{e^z}{(z-9)^2}=\dfrac{1}{81}\in\bbC\setminus\left\{0\right\}$

      Sea $g(z)=(z-0)f(z)=\dfrac{e^z}{(z-9)^2}$\\

      $\text{Res}(f,0)=\dfrac{g(0)}{0!}=\dfrac{1}{81}$\\

      $z=9$ es un polo de orden 2 $\Longrightarrow \lim_{z\rightarrow 9} (z-9)f(z)=
      \lim_{z\rightarrow 9}\dfrac{e^z}{z}=\dfrac{e^z}{z}=\dfrac{l^9}{9}\in\bbC\setminus\left\{0\right\}$

      Sea $g(z)=(z-9)^2f(z)=\dfrac{e^z}{z}$\\

      $\text{Res}(f,9)=\dfrac{g^\prime(9)}{1!}=\dfrac{9e^9-e^9}{9^2}=\dfrac{8e^9}{81}$
    }
    \ejemplo{Calcula esta integral $I=\displaystyle\int_{C(0,1)}\dfrac{e^{\nicefrac{1}{z}}}{z^2}\ dz$}{
      Sea $f(z)=\dfrac{e^{\nicefrac{1}{e}}}{z^2}$\ \ $\displaystyle\int_{C(0,1)}
      \dfrac{e^{\nicefrac{1}{z}}}{z^2}\ dz=2\pi i\ \text{Res(f,0)}$

      La singularidad $z=0$ es esencial.\\
      $f(z)=z^{-2}e^{\nicefrac{1}{z}}=z^{-2}\displaystyle\sum_{n=0}^\infty \dfrac{(\nicefrac{1}{z})^n}{n!}
      =z^{-2}\displaystyle\sum_{n=-\infty}^{0}\dfrac{z^n}{(-n)!}=\displaystyle\sum_{n=-\infty}^{0}\dfrac{z^{n-2}}{(-n)!}=
      \displaystyle\sum_{n=-\infty}^{-2}\dfrac{z^n}{-(n+2)!}\Longrightarrow \text{Res}(f,0)=a_{-1}=0$
      $\Longrightarrow I=0$
    }
    \ejemplo{$f(z)=\dfrac{1}{z(e^z-1)}\ \ \text{Res}(f,z)=0\ \forall z\neq 0$}{
      $z=0$ es un polo de orden 2 $\Longrightarrow \lim_{z\rightarrow 0}(z-0)^n f(n)=
      \lim_{z\rightarrow 0}\dfrac{z}{e^z-1}=1 \ (L'H)\in\bbC\setminus\left\{0\right\}$

      Sea $g(z)=(z-0)^2f(z)=\dfrac{z}{e^z-1}$\ Res$(f,0)=\dfrac{g^\prime(0)}{1!}$\\

      $g^\prime(z)=\dfrac{e^z-1-ze^z}{(e^z-1)^2}\Longrightarrow g^\prime(0)=
      \lim_{z\rightarrow 0}\dfrac{e^z-1-ze^z}{(e^z-1)^2}=\lim_{z\rightarrow 0}
      \dfrac{e^z-e^z-ze^z}{2(e^z-1)e^z}=-\lim_{z\rightarrow 0}\dfrac{z}{2(e^z-1)}=-\dfrac{1}{2}$

      $\Longrightarrow \text{Res}(f,0)=-\dfrac{1}{2}$
    }
    \teorema{DE LOS RESIDUOS}{
       Sea $U\subseteq \bbC$ un abierto, $S\subseteq U$ sin acumulación, y $f\colon U\setminus S\to\bbC$ analítica.
       Sea $\gamma$ un camino cerrado con $\gamma^* \subseteq U\setminus S$ homólogo a $0$ en $U$. Entonces:

       \begin{equation}
          \int_\gamma f(w)\ dw=\sum_{w\in S} n(\gamma,w)\cdot 2\pi i\cdot \text{Res}(f,w)
       \end{equation}
    }
    \ejemplo{Calcular $\displaystyle\int_{C(0,3)}\dfrac{e^z-1}{z^3+4z^2+4z}\ dz$}{
      Factorizamos el polinomio $z^3+4z^2+4z=z(z+2)^2\Longleftrightarrow 
      \displaystyle\int_{C(0,3)}\dfrac{e^z-1}{z^3+4z^2+4z}\ dz = \displaystyle
      \int_{C(0,3)\dfrac{e^z-1}{z(z+2)^2}}$

      $\text{Res}(f,z)=0\ \forall z \in\bbC\setminus\left\{0,-2\right\}$

      $z=0$ es un polo de orden 0 $\Longrightarrow \lim_{z\to 0}(z-0)f(z)=\lim_{z\to 0}
      (z-0)(e^z-1)=0\Longrightarrow \text{ nos hemos pasado }$ estamos en una singularidad
      evitable. $\lim_{z\to 0}f(z)=\frac{1}{4}\Longrightarrow \text{Res}(f,0)=0$\\

      $z=-2$ es un polo de orden $2$ $\Longrightarrow \lim_{z\to -2}(z-2)^2f(z)=\dfrac{e^-1}{-2}\in\bbC\setminus\{0\}$

      Sea $g(z)=(z+2)^2f(z)=\dfrac{e^z-1}{z}\Rightarrow g^\prime(z)=\dfrac{e^z z-e^z +1}{z^2}
      \Longrightarrow \text{Res}(f,-2)=\dfrac{g^\prime(-2)}{1!}=\dfrac{1-3e^{-2}}{4}$

      $\Longrightarrow \displaystyle\int_{C(0,3)}f(z)\ dz=2\pi i \dfrac{1-3e^{-2}}{4}=\dfrac{\pi(1-3e^{-2})}{2}i$
    }
